 16def vtreat_parameters(user_params: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
 17    """
 18    build a vtreat parameters dictionary, adding in user choices
 19
 20    :param user_params: dictionary of user choices
 21    :return: dictionary of user and default choices
 22    """
 23
 24    params = {
 25        "use_hierarchical_estimate": True,
 26        "coders": {
 27            "clean_copy",
 28            "missing_indicator",
 29            "indicator_code",
 30            "impact_code",
 31            "deviation_code",
 32            "logit_code",
 33            "prevalence_code",
 34        },
 35        "filter_to_recommended": True,
 36        "indicator_min_fraction": 0.02,
 37        "cross_validation_plan": vtreat.cross_plan.KWayCrossPlanYStratified(),
 38        "cross_validation_k": 5,
 39        "user_transforms": [],
 40        "sparse_indicators": False,
 41        "missingness_imputation": numpy.mean,
 42        "check_for_duplicate_frames": True,
 43        "error_on_duplicate_frames": False,
 44        "retain_cross_plan": True,
 45        "tunable_params": ["indicator_min_fraction"],
 46    }
 47    pkeys = set(params.keys())
 48    if user_params is not None:
 49        for k in user_params.keys():
 50            if k not in pkeys:
 51                raise KeyError("parameter key " + str(k) + " not recognized")
 52            params[k] = user_params[k]
 53    if params["error_on_duplicate_frames"]:
 54        params["check_for_duplicate_frames"] = True
 55    for k in params["tunable_params"]:
 56        if k not in pkeys:
 57            raise KeyError("tunable_params key " + str(k) + " not recognized")
 58    return params

175    def fit_transform(self, X, y=None, **fit_params):
176        """
177        fit_transform data, this is the way to fit with cross methods.
178
179        :param X: explanatory values
180        :param y: dependent values
181        :param fit_params:
182        :return: transformed data
183        """
184
185        X, orig_type = vtreat_impl.ready_data_frame(X)
186        self.check_column_names(X.columns)
187        if y is None:
188            if self.outcome_name_ is None:
189                raise ValueError(".fit_transform(X) must have outcome_name set")
190            y = numpy.asarray(X[self.outcome_name_])
191        else:
192            y = numpy.asarray(y)
193            if (self.outcome_name_ is not None) and (self.outcome_name_ in X.columns):
194                if not numpy.all(X[self.outcome_name_] == y):
195                    raise ValueError(
196                        ".fit_transform(X, y) called with y != X[outcome_name]"
197                    )
198        if not X.shape[0] == len(y):
199            raise ValueError("X.shape[0] should equal len(y)")
200        y = vtreat.util.safe_to_numeric_array(y)
201        if vtreat.util.is_bad(y).sum() > 0:
202            raise ValueError("y should not have any missing/NA/NaN values")
203        if numpy.max(y) <= numpy.min(y):
204            raise ValueError("y does not vary")
205        cross_rows = None
206        cross_plan = None
207        if self.params_["retain_cross_plan"]:
208            cross_rows = self.cross_rows_
209            cross_plan = self.cross_plan_
210        self.clear()
211        self.last_fit_x_id_ = vtreat.util.hash_data_frame(X)
212        X = vtreat_impl.pre_prep_frame(
213            X, col_list=self.var_list_, cols_to_copy=self.cols_to_copy_
214        )
215        if isinstance(y, pandas.Series):
216            y = y.reset_index(inplace=False, drop=True)
217        # model for independent transforms
218        self.plan_ = None
219        self.score_frame_ = None
220        self.plan_ = vtreat_impl.fit_numeric_outcome_treatment(
221            X=X,
222            y=y,
223            var_list=self.var_list_,
224            outcome_name=self.outcome_name_,
225            cols_to_copy=self.cols_to_copy_,
226            params=self.params_,
227            imputation_map=self.imputation_map_,
228        )
229        cross_frame = vtreat_impl.perform_transform(
230            x=X, transform=self, params=self.params_
231        )
232        if (cross_plan is None) or (cross_rows != X.shape[0]):
233            if cross_plan is not None:
234                warnings.warn(
235                    "Number of rows different than previous fit with retain_cross_plan==True"
236                )
237            cross_plan = self.params_["cross_validation_plan"].split_plan(
238                n_rows=X.shape[0],
239                k_folds=self.params_["cross_validation_k"],
240                data=X,
241                y=y,
242            )
243            cross_rows = X.shape[0]
244        # patch in cross-frame versions of complex columns such as impact
245        vtreat_impl.cross_patch_refit_y_aware_cols(
246            x=X, y=y, res=cross_frame, plan=self.plan_, cross_plan=cross_plan
247        )
248        vtreat_impl.cross_patch_user_y_aware_cols(
249            x=cross_frame,
250            y=y,
251            res=cross_frame,
252            params=self.params_,
253            cross_plan=cross_plan,
254        )
255        # use cross_frame to compute variable effects
256        self.score_frame_ = vtreat_impl.score_plan_variables(
257            cross_frame=cross_frame,
258            outcome=y,
259            plan=self.plan_,
260            params=self.params_,
261            is_classification=False,
262        )
263        if ("filter_to_recommended" in self.params_.keys()) and self.params_[
264            "filter_to_recommended"
265        ]:
266            self.set_result_restriction(
267                set(
268                    [
269                        ci
270                        for ci in self.score_frame_["variable"][
271                            self.score_frame_["recommended"]
272                        ]
273                    ]
274                )
275            )
276        cross_frame = vtreat_impl.limit_to_appropriate_columns(
277            res=cross_frame, transform=self
278        )
279        cross_frame, res_columns = vtreat_impl.back_to_orig_type_data_frame(
280            cross_frame, orig_type
281        )
282        self.last_result_columns = res_columns
283        if self.params_["retain_cross_plan"]:
284            self.cross_plan_ = cross_plan
285            self.cross_rows_ = cross_rows
286        else:
287            self.cross_plan_ = None
288            self.cross_rows_ = None
289        return cross_frame
